/**

 TODO - notes:

 * Only picks what is subscribed for (ie surface vs object)

 @class MousePickControl
 @module xeogl
 @submodule controls
 @constructor
 @param [scene] {Scene} Parent {{#crossLink "Scene"}}{{/crossLink}}.
 @param [cfg] {*} Configs
 @param [cfg.id] {String} Optional ID, unique among all components in the parent scene, generated automatically when omitted.
 @param [cfg.meta] {String:Object} Optional map of user-defined metadata to attach to this MousePickControl.
 @param [cfg.camera] {String|Camera} ID or instance of a {{#crossLink "Camera"}}Camera{{/crossLink}} to control.
 Must be within the same {{#crossLink "Scene"}}Scene{{/crossLink}} as this MousePickControl. Defaults to the
 parent {{#crossLink "Scene"}}Scene{{/crossLink}}'s default instance, {{#crossLink "Scene/camera:property"}}camera{{/crossLink}}.
 @param [cfg.active=true] {Boolean} Whether or not this MousePickControl is active.
 @param [firstPerson=false] {Boolean} Whether or not this MousePickControl is in "first person" mode.
 @extends Component
 */
(function () {

    "use strict";

    xeogl.MousePickControl = xeogl.Component.extend({

        /**
         JavaScript class name for this Component.

         @property type
         @type String
         @final
         */
        type: "xeogl.MousePickControl",

        _init: function (cfg) {

            var self = this;
            var scene = this.scene;
            var input = scene.input;
            var mousePos = [0, 0];
            var needPick = false;
            var needPickSurface = false;
            var pickedEntityId;
            var surfacePickHit = {
                entity: null,
                primIndex: null,
                bary: new Float32Array(3),
                worldPos: new Float32Array(3),
                normal: new Float32Array(3),
                canvasPos: new Float32Array(2)
            };
            var picked = false;
            var pickedSurface = false;

            this._active = false;

            var pick = function () {
                if (!needPick && !needPickSurface) {
                    return;
                }
                picked = false;
                pickedSurface = false;
                var hit;
                if (needPickSurface || self.hasSubs("hoverSurface")) {
                    hit = scene.pick({
                        pickSurface: true,
                        canvasPos: mousePos
                    });
                } else {
                    hit = viewer.pickObject(mousePos);
                }
                if (hit) {
                    picked = true;
                    if (pickedEntityId !== hit.entity.id) {
                        if (pickedEntityId !== undefined) {

                            /**
                             Fired whenever this MousePickControl hovers off an entity.
                             @event hoverOut
                             @param pickedEntityId The entity ID
                             */
                            self.fire("hoverOut", pickedEntityId);
                        }
                        self.fire("hover", hit.entity.id);
                        pickedEntityId = hit.entity.id;
                    }
                    if (hit.worldPos) {
                        surfacePickHit.entity = hit.entity;
                        surfacePickHit.primIndex = hit.primIndex;
                        surfacePickHit.bary.set(hit.bary);
                        surfacePickHit.worldPos.set(hit.worldPos);
                        surfacePickHit.normal.set(hit.normal);
                        surfacePickHit.canvasPos.set(mousePos);
                        pickedSurface = true;

                        /**
                         Fired whenever this MousePickControl hovers over the surface of an entity.
                         @event hoverSurface
                         @param hit
                         */
                        self.fire("hoverSurface", surfacePickHit);
                    }
                } else {
                    if (pickedEntityId !== undefined) {
                        self.fire("hoverOut", pickedEntityId);
                        pickedEntityId = undefined;
                    }
                    self.fire("hoverOff", {
                        canvasPos: mousePos
                    });
                }
                needPick = false;
                needPickSurface = false;
            };

            var tick = function () {
                if (self._active) {
                    pick();
                }
                requestAnimationFrame(tick);
            };

            tick();

            self._onMouseMove = input.on("mousemove", function (canvasPos) {
                if (!self._active) {
                    return;
                }
                mousePos[0] = canvasPos[0];
                mousePos[1] = canvasPos[1];
                needPick = true;
            });

            self._onClick = input.on('click', (function (e) {
                var clicks = 0;
                var timeout;
                return function (e) {
                    if (!self._active) {
                        return;
                    }
                    if (!self.hasSubs("doubleClicked") && !self.hasSubs("doubleClickedSurface") && !self.hasSubs("doubleCLickedNothing")) {
                        //  Avoid the single/double click differentiation timeout
                        needPickSurface = !!self.hasSubs("clickedSurface");
                        pick();
                        if (pickedEntityId) {

                            /**
                             Fired whenever this MousePickControl clicks the surface of an entity.
                             @event clickedSurface
                             @param hit
                             */
                            self.fire("clicked", pickedEntityId);
                            if (pickedSurface) {
                                self.fire("clickedSurface", surfacePickHit);
                            }
                        } else {

                            /**
                             Fired whenever this MousePickControl clicks empty space.
                             @event clickedNothing
                             */
                            self.fire("clickedNothing");
                        }
                        return;
                    }
                    clicks++;
                    if (clicks == 1) {
                        timeout = setTimeout(function () {
                            needPickSurface = !!self.hasSubs("clickedSurface");
                            pick();
                            if (pickedEntityId) {
                                self.fire("clicked", pickedEntityId);
                                if (pickedSurface) {
                                    self.fire("clickedSurface", surfacePickHit);
                                }
                            } else {
                                self.fire("clickedNothing");
                            }
                            clicks = 0;
                        }, 250);
                    } else {
                        clearTimeout(timeout);
                        needPickSurface = !!self.hasSubs("doubleClickedSurface");
                        pick();
                        if (pickedEntityId) {

                            /**
                             Fired whenever this MousePickControl double-clicks an entity.
                             @event clickedSurface
                             @param pickedEntityId
                             */
                            self.fire("doubleClicked", pickedEntityId);
                            if (pickedSurface) {

                                /**
                                 Fired whenever this MousePickControl double-clicks an entity surface.
                                 @event clickedSurface
                                 @param hit
                                 */
                                self.fire("doubleClickedSurface", surfacePickHit);
                            }
                        } else {

                            /**
                             Fired whenever this MousePickControl double-clicks empty space
                             @event doubleClickedNothing
                             */
                            self.fire("doubleClickedNothing");
                        }
                        clicks = 0;
                    }
                };
            })(), false);

            this.active = cfg.active;
        },

        _props: {

            /**
             * Flag which indicates whether this MousePickControl is active or not.
             *
             * Fires an {{#crossLink "MousePickControl/active:event"}}{{/crossLink}} event on change.
             *
             * @property active
             * @type Boolean
             */
            active: {

                set: function (value) {

                    value = value !== false;

                    if (this._active === value) {
                        return;
                    }

                    this._active = value;

                    /**
                     * Fired whenever this MousePickControl's {{#crossLink "MousePickControl/active:property"}}{{/crossLink}} property changes.
                     * @event active
                     * @param value The property's new value
                     */
                    this.fire('active', this._active = value);
                },

                get: function () {
                    return this._active;
                }
            }
        },

        _getJSON: function () {
            var json = {
                active: this._active
            };
            return json;
        },

        _destroy: function () {
            this.scene.input.off(this._onMouseMove);
            this.scene.input.off(this._onClick);
            this.active = false;
        }
    });

})();
